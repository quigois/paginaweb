-- Schema fraccionamiento

CREATE SCHEMA IF NOT EXISTS `fraccionamiento` DEFAULT CHARACTER SET latin1 ;

USE `fraccionamiento` ;

-- Table `fraccionamiento`.`materiales`

CREATE TABLE IF NOT EXISTS `fraccionamiento`.`materiales` (

`id_material` INT(10) NOT NULL DEFAULT '0',

`nombre` VARCHAR(15) NULL DEFAULT NULL,

`costo` INT(10) NULL DEFAULT NULL,

PRIMARY KEY (`id_material`))

ENGINE = InnoDB

DEFAULT CHARACTER SET = latin1;

-- Table `fraccionamiento`.`residente`

CREATE TABLE IF NOT EXISTS `fraccionamiento`.`residente` (

`id_residente` INT(10) NOT NULL DEFAULT '0',

`nombre` VARCHAR(15) NULL DEFAULT NULL,

`apellido_p` VARCHAR(15) NULL DEFAULT NULL,

`apellido_m` VARCHAR(15) NULL DEFAULT NULL,

`calle` VARCHAR(15) NULL DEFAULT NULL,

`numero` INT(10) NULL DEFAULT NULL,

PRIMARY KEY (`id_residente`))

ENGINE = InnoDB

DEFAULT CHARACTER SET = latin1;

-- Table `fraccionamiento`.`tipo_pago`

CREATE TABLE IF NOT EXISTS `fraccionamiento`.`tipo_pago` (

`id_pago` INT(10) NOT NULL DEFAULT '0',

`tipo` VARCHAR(10) NULL DEFAULT NULL,

PRIMARY KEY (`id_pago`))

ENGINE = InnoDB

DEFAULT CHARACTER SET = latin1;

-- Table `fraccionamiento`.`servicio`

CREATE TABLE IF NOT EXISTS `fraccionamiento`.`servicio` (

`id_servicio` INT(10) NOT NULL DEFAULT '0',

`nombre_servicio` VARCHAR(15) NULL DEFAULT NULL,

PRIMARY KEY (`id_servicio`))

ENGINE = InnoDB

DEFAULT CHARACTER SET = latin1;

-- Table `fraccionamiento`.`recibo`

CREATE TABLE IF NOT EXISTS `fraccionamiento`.`recibo` (

`id_pago` INT(10) NULL DEFAULT NULL,

`id_residente` INT(10) NULL DEFAULT NULL,

`id_servicio` INT(10) NULL DEFAULT NULL,

UNIQUE INDEX `id_pago` (`id_pago` ASC),

UNIQUE INDEX `id_residente` (`id_residente` ASC),

UNIQUE INDEX `id_servicio` (`id_servicio` ASC),

CONSTRAINT `recibo_ibfk_1`

FOREIGN KEY (`id_residente`)

REFERENCES `fraccionamiento`.`residente` (`id_residente`)

ON DELETE CASCADE ON UPDATE CASCADE,

CONSTRAINT `recibo_ibfk_2`

FOREIGN KEY (`id_pago`)

REFERENCES `fraccionamiento`.`tipo_pago` (`id_pago`)

ON DELETE CASCADE ON UPDATE CASCADE,

CONSTRAINT `recibo_ibfk_3`

FOREIGN KEY (`id_servicio`)

REFERENCES `fraccionamiento`.`servicio` (`id_servicio`)

ON DELETE CASCADE ON UPDATE CASCADE)

ENGINE = InnoDB

DEFAULT CHARACTER SET = latin1;

-- Table `fraccionamiento`.`sueldos`

CREATE TABLE IF NOT EXISTS `fraccionamiento`.`sueldos` (

`id_sueldos` INT(10) NOT NULL DEFAULT '0',

`nombre_sueldo` VARCHAR(15) NULL DEFAULT NULL,

`monto` INT(10) NULL DEFAULT NULL,

PRIMARY KEY (`id_sueldos`))

ENGINE = InnoDB

DEFAULT CHARACTER SET = latin1;
